--CREAR VISTAS
CREATE OR REPLACE VIEW vista_subestaciones_activas AS
SELECT id, nombre, voltaje, geom
FROM subestaciones_energia
WHERE estado = 'OPERATIVA';

CREATE OR REPLACE VIEW vista_subestaciones_medellin AS
SELECT 
    s.ogc_fid, 
    s.nombre_subestacion, 
    s.tension, 
    s.vigencia, 
    s.geom
FROM subestaciones_energia s
JOIN municipios m ON ST_Intersects(s.geom, m.geom)
WHERE UPPER(m.mpnombre) = 'MEDELLÍN';

SELECT ST_SRID(geom), COUNT(*)
FROM municipios
GROUP BY ST_SRID(geom);


CREATE OR REPLACE VIEW vista_lineas_medellin AS
SELECT 
    l.ogc_fid, 
    l.nombre_circuito, 
    l.tension, 
    l.geom
FROM sistema_de_transmision_energia l
JOIN municipios m ON ST_Intersects(l.geom, m.geom)
WHERE UPPER(m.mpnombre) = 'MEDELLÍN';
